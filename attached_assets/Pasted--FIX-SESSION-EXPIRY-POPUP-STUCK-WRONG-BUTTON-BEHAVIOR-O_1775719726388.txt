# 🎨 FRONTEND FIX

---

## 🔹 STEP 3: CENTRAL SESSION HANDLER

```ts id="session-handler"
let sessionTimeout;
let warningTimeout;

function startSessionTimer(expiresIn) {
  clearTimeout(sessionTimeout);
  clearTimeout(warningTimeout);

  // Show popup before expiry (e.g., 60 sec)
  warningTimeout = setTimeout(() => {
    showSessionModal();
  }, expiresIn - 60000);

  // Auto logout on expiry
  sessionTimeout = setTimeout(() => {
    handleLogout();
  }, expiresIn);
}
```

---

## 🔹 STEP 4: HANDLE "STAY LOGGED IN"

```ts id="stay-logged-in"
async function handleStayLoggedIn() {
  try {
    const res = await refreshSession();

    updateToken(res.token);
    startSessionTimer(res.expiresIn);

    closeSessionModal();
  } catch (err) {
    handleLogout(); // fallback
  }
}
```

---

## 🔹 STEP 5: HANDLE LOGOUT

```ts id="logout"
function handleLogout() {
  clearSession();
  closeSessionModal();
  redirectToLogin();
}
```

---

## 🔹 STEP 6: AUTO CLOSE MODAL ON LOGOUT

```ts id="modal-cleanup"
useEffect(() => {
  if (!isAuthenticated) {
    closeSessionModal();
  }
}, [isAuthenticated]);
```

---

## 🔹 STEP 7: HANDLE 401 GLOBALLY

```ts id="axios-interceptor"
axios.interceptors.response.use(
  res => res,
  err => {
    if (err.response?.status === 401) {
      handleLogout();
    }
    return Promise.reject(err);
  }
);
```
