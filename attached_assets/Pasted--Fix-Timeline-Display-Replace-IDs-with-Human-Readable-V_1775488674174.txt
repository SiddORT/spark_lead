## ✅ 1. Create Mapping Functions

---

### 🔹 Users Mapping

```ts
const getUserName = (id) => {
  return users.find(u => u.id === id)?.display_name || "Unknown User";
};
```

---

### 🔹 Services Mapping

```ts
const getServiceName = (id) => {
  return services.find(s => s.id === id)?.name || "Unknown Service";
};
```

---

### 🔹 Companies Mapping

```ts
const getCompanyName = (id) => {
  return companies.find(c => c.id === id)?.name || "Unknown Company";
};
```

---

### 🔹 Pipeline Status Mapping

```ts
const getStatusName = (id) => {
  return pipelineStatuses.find(p => p.id === id)?.name || "Unknown Status";
};
```

## ✅ 2. Transform Timeline Before Rendering

```ts
const formattedTimeline = timeline.map(item => {
  let value = item.new_value;

  switch (item.field) {
    case "dealHandler":
      value = getUserName(item.new_value);
      break;

    case "leadOwner":
      value = getUserName(item.new_value);
      break;

    case "serviceId":
      value = getServiceName(item.new_value);
      break;

    case "company":
      value = getCompanyName(item.new_value);
      break;

    case "pipelineStatusId":
      value = getStatusName(item.new_value);
      break;
  }

  return {
    ...item,
    displayValue: value,
  };
});
```
