# Add Companies Field — Lead Creation Form & All Lead Views

## What Needs to Be Done

Add a **Companies** multi-select field directly below the **Service** dropdown in:
1. **New Lead form** (`/leads/new`) — with auto-filtering based on selected service
2. **Lead Detail Sheet** (Details tab) — same auto-filtering behavior
3. **Dashboard leads table** — Companies column display
4. **Kanban cards** — Companies display
5. **CSV Export** — Companies included

The Companies field must be fully wired to the `company_services` and `lead_companies` junction tables. Selecting a service filters companies to only those linked to that service. Changing the service resets and re-filters companies. Companies are stored in `lead_companies` (many-to-many).

## Part 1 — Database (Verify Tables Exist)

These tables must exist. If not, create migrations:

```sql
-- lead_companies — many-to-many: leads ↔ companies
CREATE TABLE IF NOT EXISTS lead_companies (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  lead_id UUID NOT NULL REFERENCES leads(id) ON DELETE CASCADE,
  company_id UUID NOT NULL REFERENCES companies(id) ON DELETE CASCADE,
  UNIQUE(lead_id, company_id)
);

-- company_services — many-to-many: companies ↔ services
CREATE TABLE IF NOT EXISTS company_services (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  company_id UUID NOT NULL REFERENCES companies(id) ON DELETE CASCADE,
  service_id UUID NOT NULL REFERENCES services(id) ON DELETE CASCADE,
  UNIQUE(company_id, service_id)
);
```

// Body: { companyIds: string[] }
// Replaces all company links for a lead (delete existing, insert new)
router.post('/leads/:leadId/companies', requireAuth, requirePermission('leads', 'update'), async (req, res) => {
  const { leadId } = req.params;
  const { companyIds } = req.body as { companyIds: string[] };

  await db.transaction(async (tx) => {
    // Remove all existing links for this lead
    await tx.delete(leadCompanies).where(eq(leadCompanies.leadId, leadId));
    // Insert new links
    if (companyIds.length > 0) {
      await tx.insert(leadCompanies).values(
        companyIds.map(companyId => ({ leadId, companyId }))
      );
    }
  });

  res.json({ success: true });
});
```

export const useAllCompanies = () => {

      const res = await apiCall('/companies');
      return res.json() as Promise<{ id: string; name: string; industry: string }[]>;
    },
  });

// Fetch companies filtered by serviceId
// Returns empty array (not all companies) when no service selected
export const useCompaniesByService = (serviceId: string | null) => {
  return useQuery({
    queryKey: ['companies', 'by-service', serviceId],
    queryFn: async () => {
      if (!serviceId) return [];
      const res = await apiCall(`/companies/by-service/${serviceId}`);
      return res.json() as Promise<{ id: string; name: string }[]>;
    },
    enabled: !!serviceId,
  });
};

// Fetch companies linked to a specific lead
export const useLeadCompanies = (leadId: string | null) => {
  return useQuery({
    queryKey: ['lead-companies', leadId],
    queryFn: async () => {
      if (!leadId) return [];
      const res = await apiCall(`/leads/${leadId}/companies`);
      return res.json() as Promise<{ id: string; name: string }[]>;
    },
    enabled: !!leadId,
  });
};

// Mutation: save company links for a lead
export const useSaveLeadCompanies = () => {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: async ({ leadId, companyIds }: { leadId: string; companyIds: string[] }) => {
      const res = await apiCall(`/leads/${leadId}/companies`, {
        method: 'POST',
        body: JSON.stringify({ companyIds }),
      });
      return res.json();
    },
    onSuccess: (_, { leadId }) => {
      queryClient.invalidateQueries({ queryKey: ['lead-companies', leadId] });
      queryClient.invalidateQueries({ queryKey: ['leads'] });
    },
  });
};
```

## Part 4 — CompaniesMultiSelect Component

Create a shared, reusable component used in both the New Lead form and the Lead Detail Sheet:

```tsx
// components/CompaniesMultiSelect.tsx
import { useState } from 'react';
import { Building2, Check, ChevronDown, X } from 'lucide-react';

interface Company { id: string; name: string; }

interface Props {
  serviceId: string | null;
  selectedIds: string[];
  onChange: (ids: string[]) => void;
  disabled?: boolean;
}

export const CompaniesMultiSelect = ({ serviceId, selectedIds, onChange, disabled }: Props) => {
  const [isOpen, setIsOpen] = useState(false);
  const { data: availableCompanies = [], isLoading } = useCompaniesByService(serviceId);

  const toggleCompany = (id: string) => {
    if (selectedIds.includes(id)) {
      onChange(selectedIds.filter(i => i !== id));
    } else {
      onChange([...selectedIds, id]);
    }
  };

  const removeCompany = (id: string, e: React.MouseEvent) => {
    e.stopPropagation();
    onChange(selectedIds.filter(i => i !== id));
  };

  const selectedCompanies = availableCompanies.filter(c => selectedIds.includes(c.id));

  // State: no service selected
  if (!serviceId) {
    return (
      <div className="companies-select-empty">
        <Building2 size={14} />
        <span>Select a service first to see linked companies</span>
      </div>
    );
  }

  // State: service selected but loading
  if (isLoading) {
    return (
      <div className="companies-select-empty">
        <span className="spinner-xs" />
        <span>Loading companies…</span>
      </div>
    );
  }

  // State: service selected but no companies linked
  if (availableCompanies.length === 0) {
    return (
      <div className="companies-select-empty companies-select-warning">
        <Building2 size={14} />
        <span>No companies linked to this service. <a href="/master/services">Link companies →</a></span>
      </div>
    );
  }

  return (
    <div className="companies-multiselect" style={{ position: 'relative' }}>
      {/* Trigger */}
      <div
        className={`companies-trigger ${disabled ? 'disabled' : ''} ${isOpen ? 'open' : ''}`}
        onClick={() => !disabled && setIsOpen(!isOpen)}
        role="button"
        tabIndex={0}
        onKeyDown={e => e.key === 'Enter' && !disabled && setIsOpen(!isOpen)}
      >
        {selectedCompanies.length === 0 ? (
          <span className="companies-placeholder">Select companies…</span>
        ) : (
          <div className="companies-chips">
            {selectedCompanies.map(company => (
              <span key={company.id} className="company-chip">
                {company.name}
                <button
                  className="company-chip-remove"
                  onClick={(e) => removeCompany(company.id, e)}
                  type="button"
                  tabIndex={-1}
                >
                  <X size={10} />
                </button>
              </span>
            ))}
          </div>
        )}
        <ChevronDown
          size={14}
          className="companies-chevron"
          style={{ transform: isOpen ? 'rotate(180deg)' : 'rotate(0deg)', transition: 'transform 150ms ease' }}
        />
      </div>

      {/* Dropdown panel */}
      {isOpen && (
        <>
          {/* Backdrop */}
          <div
            style={{ position: 'fixed', inset: 0, zIndex: 998 }}
            onClick={() => setIsOpen(false)}
          />
          <div className="companies-dropdown">
            <div className="companies-dropdown-header">
              <span>{availableCompanies.length} compan{availableCompanies.length === 1 ? 'y' : 'ies'} available</span>
              {selectedIds.length > 0 && (
                <button
                  className="companies-clear-all"
                  onClick={() => onChange([])}
                  type="button"
                >
                  Clear all
                </button>
              )}
            </div>
            <div className="companies-dropdown-list">
              {availableCompanies.map(company => {
                const isSelected = selectedIds.includes(company.id);
                return (
                  <div
                    key={company.id}
                    className={`companies-option ${isSelected ? 'selected' : ''}`}
                    onClick={() => toggleCompany(company.id)}
                    role="option"
                    aria-selected={isSelected}
                  >
                    <div className={`companies-option-check ${isSelected ? 'checked' : ''}`}>
                      {isSelected && <Check size={10} strokeWidth={3} />}
                    </div>
                    <span className="companies-option-name">{company.name}</span>
                  </div>
                );
              })}
            </div>
          </div>
        </>
      )}
    </div>
  );
};
```
