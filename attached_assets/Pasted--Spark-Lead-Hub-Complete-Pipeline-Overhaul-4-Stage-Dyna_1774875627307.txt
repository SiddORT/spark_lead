# 🔄 Spark Lead Hub — Complete Pipeline Overhaul: 4-Stage + Dynamic Status System

This is a **full pipeline architecture replacement**. The old `discovery → qualification → strategy → resolution` system with its hardcoded qualify/strategy/resolve tabs is being completely replaced with a flexible, master-data-driven 4-stage pipeline with dynamic per-stage statuses. Every single part of the system that touches stage or status must be updated.

**Do not leave any trace of the old stage names anywhere — not in the DB, not in the UI, not in enums, not in filter dropdowns, not in charts, not in CSV exports, not in audit logs.**

---

## THE NEW PIPELINE (Source of Truth)

```
Stage 1: Lead Initiation
  → Statuses: New Lead Connected

Stage 2: Qualification & Analysis
  → Statuses: Internal Discussion, Requirement Analysis, Eligibility Check

Stage 3: Proposal & Negotiation
  → Statuses: Proposal Shared, Ongoing Negotiation

Stage 4: Closure
  → Statuses: Closed – Won, Closed – Lost, Closed – Postponed
```

```sql
-- ─── pipeline_stages — master table for stages ───
CREATE TABLE pipeline_stages (
  id           UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name         TEXT NOT NULL UNIQUE,
  display_name TEXT NOT NULL,
  description  TEXT,
  color        TEXT NOT NULL DEFAULT '#6b7a9a',
  icon         TEXT,
  sort_order   INTEGER NOT NULL DEFAULT 0,
  is_active    BOOLEAN NOT NULL DEFAULT true,
  is_terminal  BOOLEAN NOT NULL DEFAULT false, -- true for Closure stage
  created_at   TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at   TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- ─── pipeline_statuses — master table for statuses ───
CREATE TABLE pipeline_statuses (
  id           UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  stage_id     UUID NOT NULL REFERENCES pipeline_stages(id) ON DELETE CASCADE,
  name         TEXT NOT NULL,
  display_name TEXT NOT NULL,
  description  TEXT,
  color        TEXT NOT NULL DEFAULT '#6b7a9a',
  sort_order   INTEGER NOT NULL DEFAULT 0,
  is_active    BOOLEAN NOT NULL DEFAULT true,
  is_won       BOOLEAN NOT NULL DEFAULT false,  -- for "Closed – Won"
  is_lost      BOOLEAN NOT NULL DEFAULT false,  -- for "Closed – Lost" / "Closed – Postponed"
  created_at   TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at   TIMESTAMPTZ NOT NULL DEFAULT now(),
  UNIQUE(stage_id, name)
);

-- ─── Add stage_id and status_id to leads table ───
ALTER TABLE leads
  ADD COLUMN IF NOT EXISTS pipeline_stage_id   UUID REFERENCES pipeline_stages(id),
  ADD COLUMN IF NOT EXISTS pipeline_status_id  UUID REFERENCES pipeline_statuses(id);

-- Keep old 'stage' column temporarily for migration, then drop after migration confirmed
-- ALTER TABLE leads DROP COLUMN stage;         ← run AFTER migration verified
-- ALTER TABLE leads DROP COLUMN outcome;       ← run AFTER migration verified

-- Update triggers for updated_at
CREATE OR REPLACE FUNCTION update_updated_at()
RETURNS TRIGGER AS $$
BEGIN NEW.updated_at = now(); RETURN NEW; END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER pipeline_stages_updated_at
  BEFORE UPDATE ON pipeline_stages
  FOR EACH ROW EXECUTE FUNCTION update_updated_at();

CREATE TRIGGER pipeline_statuses_updated_at
  BEFORE UPDATE ON pipeline_statuses
  FOR EACH ROW EXECUTE FUNCTION update_updated_at();
```

```ts
router.post('/leads', requireAuth, requirePermission('leads', 'create'), async (req, res) => {
  const { companyIds = [], pipelineStageId, pipelineStatusId, ...leadData } = req.body;

  // Validate that status belongs to stage
  if (pipelineStageId && pipelineStatusId) {
    const status = await db.query.pipelineStatuses.findFirst({
      where: and(
        eq(pipelineStatuses.id, pipelineStatusId),
        eq(pipelineStatuses.stageId, pipelineStageId)
      )
    });
    if (!status) {
      return res.status(400).json({ error: 'Status does not belong to the selected stage' });
    }
  }

  const [newLead] = await db.insert(leads).values({
    ...leadData,
    pipelineStageId,
    pipelineStatusId,
    createdBy: req.user!.userId,
  }).returning();

  // Link companies
  if (companyIds.length > 0) {
    await db.insert(leadCompanies).values(companyIds.map((cid: string) => ({ leadId: newLead.id, companyId: cid })));
  }

  // Log activity
  await logActivities(newLead.id, req.user!.userId, [{
    action: 'created', fieldName: 'lead', oldValue: null, newValue: newLead.leadName
  }]);

  // Auto-set resolvedAt if stage is terminal
  const stage = await db.query.pipelineStages.findFirst({ where: eq(pipelineStages.id, pipelineStageId) });
  if (stage?.isTerminal && !newLead.resolvedAt) {
    await db.update(leads).set({ resolvedAt: new Date() }).where(eq(leads.id, newLead.id));
  }

  res.status(201).json(newLead);
});

// PATCH /api/leads/:id — handle pipeline stage/status change
router.patch('/leads/:id', requireAuth, requirePermission('leads', 'update'), async (req, res) => {
  const { pipelineStageId, pipelineStatusId, ...otherUpdates } = req.body;
  const updates: Record<string, any> = { ...otherUpdates, updatedAt: new Date() };

  // Validate status belongs to stage if both provided
  if (pipelineStageId && pipelineStatusId) {
    const status = await db.query.pipelineStatuses.findFirst({
      where: and(
        eq(pipelineStatuses.id, pipelineStatusId),
        eq(pipelineStatuses.stageId, pipelineStageId)
      )
    });
    if (!status) {
      return res.status(400).json({ error: 'Status does not belong to the selected stage' });
    }
    updates.pipelineStageId  = pipelineStageId;
    updates.pipelineStatusId = pipelineStatusId;
  } else if (pipelineStageId && !pipelineStatusId) {
    // If only stage changed, auto-select first status of new stage
    const firstStatus = await db.query.pipelineStatuses.findFirst({
      where: and(eq(pipelineStatuses.stageId, pipelineStageId), eq(pipelineStatuses.isActive, true)),
      orderBy: asc(pipelineStatuses.sortOrder),
    });
    updates.pipelineStageId  = pipelineStageId;
    updates.pipelineStatusId = firstStatus?.id || null;
  }

  // Auto-set resolvedAt when moved to terminal stage for the first time
  if (pipelineStageId) {
    const stage = await db.query.pipelineStages.findFirst({ where: eq(pipelineStages.id, pipelineStageId) });
    const existingLead = await db.query.leads.findFirst({ where: eq(leads.id, req.params.id) });
    if (stage?.isTerminal && !existingLead?.resolvedAt) {
      updates.resolvedAt = new Date();
    }
    if (!stage?.isTerminal) {
      updates.resolvedAt = null; // clear if moved back from terminal
    }
  }

  // Log stage/status changes
  const existing = await db.query.leads.findFirst({ where: eq(leads.id, req.params.id) });
  const activities: ActivityLog[] = [];
  if (pipelineStageId && pipelineStageId !== existing?.pipelineStageId) {
    activities.push({ fieldName: 'stage', oldValue: existing?.stageName, newValue: updates.pipelineStageId });
  }
  if (pipelineStatusId && pipelineStatusId !== existing?.pipelineStatusId) {
    activities.push({ fieldName: 'status', oldValue: existing?.statusName, newValue: updates.pipelineStatusId });
  }

  const [updated] = await db.update(leads).set(updates).where(eq(leads.id, req.params.id)).returning();

  if (activities.length > 0) await logActivities(req.params.id, req.user!.userId, activities);

  // Send email notification on stage change
  if (pipelineStageId !== existing?.pipelineStageId) {
    sendActivityAlertEmail({ ... }).catch(console.error);
  }

  res.json(updated);
});
```

## PART 4 — StageStatusSelect Component (Shared, Reusable)

Used in New Lead form, Lead Detail Sheet, and anywhere stage/status is shown as editable:

```tsx
// components/StageStatusSelect.tsx

interface Props {
  stageId: string | null;
  statusId: string | null;
  onStageChange: (stageId: string, autoStatus: string) => void;
  onStatusChange: (statusId: string) => void;
  disabled?: boolean;
  size?: 'sm' | 'md';
}

export const StageStatusSelect = ({
  stageId, statusId, onStageChange, onStatusChange, disabled, size = 'md'
}: Props) => {
  const { data: stages = [], isLoading } = usePipelineStages();
  const currentStage = stages.find(s => s.id === stageId);
  const availableStatuses = currentStage?.statuses ?? [];

  const handleStageChange = (newStageId: string) => {
    const newStage = stages.find(s => s.id === newStageId);
    const firstStatus = newStage?.statuses[0];
    // Auto-select first status when stage changes
    onStageChange(newStageId, firstStatus?.id ?? '');
  };

  if (isLoading) return <div className="pipeline-select-loading">Loading pipeline…</div>;

  return (
    <div className="stage-status-select">
      {/* Stage Dropdown */}
      <div className="form-field">
        <label className="field-label">Stage</label>
        <select
          className="field-select"
          value={stageId || ''}
          onChange={e => handleStageChange(e.target.value)}
          disabled={disabled}
        >
          <option value="" disabled>Select stage…</option>
          {stages.map(stage => (
            <option key={stage.id} value={stage.id}>
              {stage.displayName}
            </option>
          ))}
        </select>
        {currentStage && (
          <span className="field-helper" style={{ color: currentStage.color }}>
            {currentStage.description}
          </span>
        )}
      </div>

      {/* Status Dropdown — only visible when stage is selected */}
      {stageId && (
        <div className="form-field">
          <label className="field-label">Status</label>
          {availableStatuses.length === 0 ? (
            <div className="pipeline-no-statuses">No statuses configured for this stage</div>
          ) : (
            <select
              className="field-select"
              value={statusId || ''}
              onChange={e => onStatusChange(e.target.value)}
              disabled={disabled}
            >
              <option value="" disabled>Select status…</option>
              {availableStatuses.map(status => (
                <option key={status.id} value={status.id}>
                  {status.displayName}
                </option>
              ))}
            </select>
          )}
        </div>
      )}
    </div>
  );
};
```
